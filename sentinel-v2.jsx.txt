function Bar({ val, max = 100, color }) {
  return (
    <div style={{ height: 3, background: "#0f1f1a", borderRadius: 2, overflow: "hidden", width: "100%" }}>

}

/* ── RADAR ZONE MAP ── */
function RadarMap({ events }) {
  const [angle, setAngle] = useState(0);
  useEffect(() => {
    const id = setInterval(() => setAngle(a => (a + 2) % 360), 30);
    return () => clearInterval(id);
  }, []);

  const zoneCounts = {};
  ZONES.forEach(z => zoneCounts[z.id] = 0);
  events.filter(e => !e.resolved).forEach(e => {
    const z = ZONES.find(z => z.name === e.zone || z.id === e.zone);
    if (z) zoneCounts[z.id]++;
  });

  const rad = angle * Math.PI / 180;

  return (
    <div style={{ position: "relative", width: "100%", paddingBottom: "100%", overflow: "hidden" }}>
      <svg viewBox="0 0 200 200" style={{ position: "absolute", inset: 0, width: "100%", height: "100%" }}>
        {/* Grid circles */}
        {[20, 40, 60, 80].map(r => (
          <circle key={r} cx="100" cy="100" r={r} fill="none" stroke={C.border} strokeWidth="0.5" />
        ))}
        {/* Cross-hairs */}
        <line x1="100" y1="20" x2="100" y2="180" stroke={C.border} strokeWidth="0.5" />
        <line x1="20" y1="100" x2="180" y2="100" stroke={C.border} strokeWidth="0.5" />
        {/* Radar sweep */}
        <defs>
          <radialGradient id="sweep" cx="50%" cy="50%">
            <stop offset="0%" stopColor={C.accent} stopOpacity="0.35" />
            <stop offset="100%" stopColor={C.accent} stopOpacity="0" />
          </radialGradient>
        </defs>
        <path
          d={`M 100 100 L ${100 + 85 * Math.cos(rad)} ${100 + 85 * Math.sin(rad)} A 85 85 0 0 0 ${100 + 85 * Math.cos(rad - 0.5)} ${100 + 85 * Math.sin(rad - 0.5)} Z`}
          fill="url(#sweep)" />
        <line x1="100" y1="100" x2={100 + 85 * Math.cos(rad)} y2={100 + 85 * Math.sin(rad)} stroke={C.accent} strokeWidth="0.8" opacity="0.8" />

        {/* Zones */}
        {ZONES.map(z => {
          const cx = (z.x / 100) * 160 + 20;
          const cy = (z.y / 100) * 160 + 20;
          const count = zoneCounts[z.id] || 0;
          const color = count === 0 ? C.muted : count === 1 ? C.blue : count === 2 ? C.amber : C.red;
          return (
            <g key={z.id}>
              {count > 0 && <circle cx={cx} cy={cy} r={8 + count * 2} fill={color} opacity="0.1" />}
              <circle cx={cx} cy={cy} r={4} fill={color} opacity={0.9} />
              <text x={cx} y={cy - 7} textAnchor="middle" fontSize="5" fill={color} fontFamily="Share Tech Mono, monospace">{z.id}</text>
              {count > 0 && <text x={cx + 6} y={cy + 2} fontSize="6" fill={color} fontFamily="Share Tech Mono">{count}</text>}
            </g>
          );
        })}
        {/* Center */}
        <circle cx="100" cy="100" r="3" fill={C.accent} />
        <text x="104" y="96" fontSize="5" fill={C.accent} fontFamily="Share Tech Mono">MCD HQ</text>
      </svg>
    </div>
  );
}
